// Seção de Declarações das variáveis 
area, largura, comprimento: real


Inicio
// Seção de Comandos, procedimento, funções, operadores, etc... 
 escreva ("Insira a largura da sala: ")
 leia (largura)
 escreva ("Insira o comprimento da sala: ")
 leia (comprimento)

 escreva ("A área de sua sala é: ", area)
